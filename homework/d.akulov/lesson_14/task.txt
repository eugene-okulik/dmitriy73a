from students s
inner join books b
on s.id = b.taken_by_student_id
inner join marks m
on s.id = m.student_id
INNER join `groups` g
on s.group_id = g.id
INNER join lessons l
on m.lesson_id = l.id

on l.subject_id = s2.id
